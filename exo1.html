SELECT ?tennis ?nom_de_famille ?nom_de_familleLabel ?prénom ?prénomLabel ?distinction_reçue 
            ?distinction_reçueLabel ?victoire ?victoireLabel ?image ?pays_de_citoyenneté ?pays_de_citoyennetéLabel ?langues_parlées__écrites_ou_signées 
            ?langues_parlées__écrites_ou_signéesLabel ?hauteur ?conjoint_ou_conjointe ?conjoint_ou_conjointeLabel ?résidence ?résidenceLabel
             ?fin_de_la_période_d_activité WHERE {
                SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE],en". }
                
              
                OPTIONAL { ?tennis wdt:P735 ?prénom. }
                OPTIONAL { ?tennis wdt:P166 ?distinction_reçue. }
                OPTIONAL { ?tennis wdt:P2522 ?victoire. }
                OPTIONAL { ?tennis wdt:P18 ?image. }
                OPTIONAL { ?tennis wdt:P27 ?pays_de_citoyenneté. }
                OPTIONAL { ?tennis wdt:P1412 ?langues_parlées__écrites_ou_signées. }
                OPTIONAL { ?tennis wdt:P2048 ?hauteur. }
                OPTIONAL { ?tennis wdt:P26 ?conjoint_ou_conjointe. }
                OPTIONAL { ?tennis wdt:P551 ?résidence. }
                OPTIONAL { ?tennis wdt:P2032 ?fin_de_la_période_d_activité. }
              }
              LIMIT 100
